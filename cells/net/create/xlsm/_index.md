```cs
// Create Workbook object.
Workbook wkb = new Workbook();

// Select default added "Sheet1".
Worksheet sht = wkb.Worksheets[0];

// Get "A1" cell in the sheet.
Cell cll = sht.Cells["A1"];

// Input the "relevant cell content" text into the "A1" cell.
cll.PutValue("relevant content!");

// Save the Excel as .xlsm file.
wkb.Save("created_one.xlsm", SaveFormat.Xlsm);
```
