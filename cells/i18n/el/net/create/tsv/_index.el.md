```cs

// Δημιουργία παρουσίας κλάσης βιβλίου εργασίας.
Workbook wkb = new Workbook();

// Πρόσβαση στο πρώτο φύλλο εργασίας του βιβλίου εργασίας.
Worksheet sht = wkb.Worksheets[0];

// Λάβετε τα επιθυμητά κελιά του φύλλου εργασίας.
Cell c00 = sht.Cells["A1"];
Cell c01 = sht.Cells["B1"];
Cell c10 = sht.Cells["A2"];
Cell c11 = sht.Cells["B2"];

// εισάγετε την τιμή στα κελιά.
c00.PutValue("ColumnA");
c01.PutValue("ColumnB");
c10.PutValue("ValueA");
c11.PutValue("ValueB");

// Αποθηκεύστε το βιβλίο εργασίας ως αρχείο .tsv.
wkb.Save("created_one.tsv");


```
