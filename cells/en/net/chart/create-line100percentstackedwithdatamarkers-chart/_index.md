```csharp
using System;
using Aspose.Cells;
using Aspose.Cells.Charts;

namespace Line100PercentStackedWithDataMarkersDemo
{
    class Program
    {
        static void Main()
        {
            // 1. Create a new workbook.
            Workbook workbook = new Workbook();

            // 2. Access the first worksheet.
            Worksheet sheet = workbook.Worksheets[0];
            Cells cells = sheet.Cells;

            // -------------------------------------------------
            // 3. Fill the worksheet with sample data.
            // -------------------------------------------------
            // Header row
            cells["A1"].Value = "Month";
            cells["B1"].Value = "Product A";
            cells["C1"].Value = "Product B";
            cells["D1"].Value = "Product C";

            // Data rows (Jan‑Jun)
            string[] months = { "Jan", "Feb", "Mar", "Apr", "May", "Jun" };
            double[,] values = {
                { 30, 20, 10 },
                { 35, 25, 15 },
                { 40, 30, 20 },
                { 45, 35, 25 },
                { 50, 40, 30 },
                { 55, 45, 35 }
            };

            for (int i = 0; i < months.Length; i++)
            {
                cells[i + 1, 0].Value = months[i];                 // Column A – Month
                cells[i + 1, 1].Value = values[i, 0];             // Column B – Product A
                cells[i + 1, 2].Value = values[i, 1];             // Column C – Product B
                cells[i + 1, 3].Value = values[i, 2];             // Column D – Product C
            }

            // -------------------------------------------------
            // 4. Add a chart to the worksheet.
            // -------------------------------------------------
            int chartIndex = sheet.Charts.Add(ChartType.Line100PercentStackedWithDataMarkers, 8, 0, 24, 8);
            Chart chart = sheet.Charts[chartIndex];

            // 5. Set chart title.
            chart.Title.Text = "Sales Trend – 100% Stacked Line with Markers";

            // 6. Add data series – the data range includes the header row.
            //    Series are added automatically for each column after the first (the categories column).
            chart.NSeries.Add("=Sheet1!$B$1:$D$7", true);

            // 7. Enable data markers for each series.
            foreach (NSeries series in chart.NSeries)
            {
                // Use default marker style (a circle) – you can change the type/color if required.
                series.Marker.IsAutoGenerated = true;
                series.Marker.Size = 7; // Marker size (points)
            }

            // 8. Set the category (X‑axis) labels to be the month names.
            chart.CategoryAxis[0].CategoryData = "=Sheet1!$A$2:$A$7";

            // 9. Optionally customize axis titles.
            chart.CategoryAxis[0].Title.Text = "Month";
            chart.ValueAxis[0].Title.Text = "Percentage";

            // 10. Save the workbook.
            workbook.Save("Line100PercentStackedWithDataMarkers.xlsx");
        }
    }
}
```
