// создать экземпляр объекта Workbook с защищенным файлом XLSX
var workbook = new Aspose.Cells.Workbook("protected.xlsx");

// получить доступ к рабочему листу по умолчанию в файле Excel
var worksheet = workbook.Worksheets[0];

// снять защиту листа без пароля
worksheet.Unprotect();

// снять защиту книги паролем
workbook.Unprotect("password");

// сохранить результат обратно в формате XLSX
workbook.Save("unprotected.xlsx", Aspose.Cells.SaveFormat.Auto);
