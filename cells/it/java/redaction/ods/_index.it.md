```cs
Workbook workbook = new Workbook(dataDir + "sourceFile.ods");

Worksheet worksheet = workbook.getWorksheets().get(0);

// Specifica l'intervallo in cui desideri eseguire la ricerca
// Qui la gamma è E3:H6
CellArea area = CellArea.createCellArea("E3", "H6");

// Specifica le opzioni di ricerca
FindOptions opts = new FindOptions();
opts.setLookInType(LookInType.VALUES);
opts.setLookAtType(LookAtType.ENTIRE_CONTENT);
opts.setRange(area);

Cell cell = null;

do {
	// Cerca la cella con la ricerca del valore all'interno dell'intervallo
	cell = worksheet.getCells().find("search", cell, opts);

	// Se non viene trovata alcuna cella di questo tipo, interrompi il ciclo
	if (cell == null)
		break;

	// Sostituisci la cella con il valore sostituisci
	cell.putValue("replace");

} while (true);

// Salva la cartella di lavoro
workbook.save(dataDir + "output.ods");


```
