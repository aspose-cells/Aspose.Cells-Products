```cs

import jpype
import asposecells
jpype.startJVM()
from asposecells.api import Workbook, FileFormatType

// إنشاء كائن المصنف.
workbook = Workbook(FileFormatType.XLTM)

// قم بالوصول إلى ورقة العمل الأولى من المصنف.
worksheet = workbook.getWorksheets().get(0)

// احصل على الخلية (الخلايا) المطلوبة من ورقة العمل وأدخل القيمة في الخلية (الخلايا).
worksheet.getCells().get("A1").putValue("ColumnA")
worksheet.getCells().get("B1").putValue("ColumnB")
worksheet.getCells().get("A2").putValue("ValueA")
worksheet.getCells().get("B2").putValue("ValueB")

// احفظ المصنف كملف XLTM.
workbook.save("output.xltm")

jpype.shutdownJVM()


```
