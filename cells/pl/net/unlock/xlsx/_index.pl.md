// utworzyć wystąpienie obiektu Workbook z chronionym plikiem XLSX
var workbook = new Aspose.Cells.Workbook("protected.xlsx");

// uzyskać dostęp do domyślnego arkusza roboczego w pliku Excel
var worksheet = workbook.Worksheets[0];

// odblokuj arkusz roboczy bez hasła
worksheet.Unprotect();

// odbezpiecz skoroszyt hasłem
workbook.Unprotect("password");

// zapisz wynik z powrotem w formacie XLSX
workbook.Save("unprotected.xlsx", Aspose.Cells.SaveFormat.Auto);
