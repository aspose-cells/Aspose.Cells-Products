    // trích xuất hình ảnh từ Trang tính 
    // mở một tệp Excel mẫu
    Workbook workbook = new Workbook("sampleExtractImagesFromWorksheets.xlsm");
    
    // lấy trang tính đầu tiên
    Worksheet worksheet = workbook.Worksheets[0];
    
    // lấy Hình ảnh đầu tiên trong trang tính đầu tiên
    Aspose.Cells.Drawing.Picture pic = worksheet.Pictures[0];
    
    // đặt đường dẫn tệp hình ảnh đầu ra
    string picformat = pic.ImageType.ToString();

    // xác định ImageOrPrintOptions

    // chỉ định định dạng hình ảnh
    printoption.ImageType =  Aspose.Cells.Drawing.ImageType.Jpeg;
                
    // lưu hình ảnh
