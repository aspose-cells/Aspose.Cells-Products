string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// tải tệp ODS trong một phiên bản của Workbook
var book = new Aspose.Cells.Workbook("template.ods");
// tạo một phiên bản của HtmlSaveOptions và đặt thuộc tính ExportHeadings thành true
var options = new Aspose.Cells.HtmlSaveOptions();
options.ExportHeadings = true;

// lưu tệp ODS ở định dạng HTML
book.Save(output, options);
