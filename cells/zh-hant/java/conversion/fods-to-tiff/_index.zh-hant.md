Workbook workbook = new Workbook("sourceFile.fods");
// 從集合中訪問默認工作表

// 定義結果圖像的參數

options.setOnePagePerSheet(true);
options.setImageType(ImageType.TIFF);
// 將工作表轉換為 TIFF 格式的圖像
SheetRender renderer = new SheetRender(worksheet, options);
renderer.toImage(0, "output.tiff");
