```cs

// instanciar um objeto Workbook com arquivo XLSX protegido
var workbook = new Aspose.Cells.Workbook("protected.xlsx");

// acessar a planilha padrão no arquivo Excel
var worksheet = workbook.Worksheets[0];

// desproteger planilha sem senha
worksheet.Unprotect();

// desproteger pasta de trabalho com senha
workbook.Unprotect("password");

workbook.Save("unprotected.xlsx", Aspose.Cells.SaveFormat.Auto);
```
