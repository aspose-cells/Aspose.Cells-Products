from aspose.cells import Workbook, SaveFormat, FileFormatType

# Create Workbook object.
workbook = Workbook()

# Access the first worksheet of the workbook.
worksheet = workbook.worksheets[0]

# Get the desired cell(s) of the worksheet and input the value into the cell(s).
worksheet.cells.get("A1").put_value("ColumnA")

# Save the workbook as GIF file.
workbook.save("output.gif")

        {{< blocks/products/pf/agp/i18n/about-file-text fileFormat="GIF" readMoreLink="https://docs.fileformat.com/image/gif/" >}}Un formato GIF o Graphical Interchange Format è un tipo di immagine altamente compressa. Di proprietà di Unisys, GIF utilizza l'algoritmo di compressione LZW che non degrada la qualità dell'immagine. Per ciascuna immagine GIF in genere consente fino a 8 bit per pixel e sono consentiti fino a 256 colori nell'immagine. Al contrario di un'immagine JPEG, che può visualizzare fino a 16 milioni di colori e raggiunge abbastanza i limiti dell'occhio umano. Ai tempi in cui è emerso Internet, le GIF sono rimaste la scelta migliore perché richiedevano una larghezza di banda ridotta e compatibilità con la grafica che consuma aree di colore solido. Un GIF animato combina numerose immagini o fotogrammi in un unico file e li visualizza in sequenza per generare una clip animata o un breve video. Le limitazioni di colore arrivano fino a 256 per ciascun fotogramma e probabilmente sono le meno adatte per riprodurre altre immagini e fotografie con gradiente di colore.{{< /blocks/products/pf/agp/i18n/about-file-text >}}
