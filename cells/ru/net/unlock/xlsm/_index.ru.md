var workbook = new Aspose.Cells.Workbook("protected.xlsm");

// получить доступ к рабочему листу по умолчанию в файле Excel

worksheet.Unprotect();

// снять защиту книги паролем

// сохранить результат обратно в формате XLSM
workbook.Save("unprotected.xlsm", Aspose.Cells.SaveFormat.Auto);
