    // estrarre le immagini dai fogli di lavoro 
    // aprire un modello di file Excel
    Workbook workbook = new Workbook("sampleExtractImagesFromWorksheets.ods");
    
    // ottenere il primo foglio di lavoro
    Worksheet worksheet = workbook.Worksheets[0];
    
    // ottenere la prima immagine nel primo foglio di lavoro
    Aspose.Cells.Drawing.Picture pic = worksheet.Pictures[0];
    
    // impostare il percorso del file immagine di output
    string picformat = pic.ImageType.ToString();
                
    // Nota: puoi valutare il formato dell'immagine prima di specificare il percorso dell'immagine
    // definire ImageOrPrintOptions
    Aspose.Cells.Rendering.ImageOrPrintOptions printoption = new  Aspose.Cells.Rendering.ImageOrPrintOptions();
    
    // specificare il formato dell'immagine
    printoption.ImageType =  Aspose.Cells.Drawing.ImageType.Jpeg;
                
    // salva l'immagine
    pic.ToImage("outputExtractImagesFromWorksheets.jpg", printoption);
