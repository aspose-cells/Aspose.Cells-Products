```cs
Workbook workbook = new Workbook(dataDir + "sourceFile.xlsm");

Worksheet worksheet = workbook.getWorksheets().get(0);

// Καθορίστε το εύρος στο οποίο θέλετε να πραγματοποιήσετε αναζήτηση
// Εδώ το εύρος είναι E3:H6
CellArea area = CellArea.createCellArea("E3", "H6");

// Καθορίστε τις επιλογές Εύρεση
FindOptions opts = new FindOptions();
opts.setLookInType(LookInType.VALUES);
opts.setLookAtType(LookAtType.ENTIRE_CONTENT);
opts.setRange(area);

Cell cell = null;

do {
	// Αναζητήστε το κελί με αναζήτηση τιμών εντός εύρους
	cell = worksheet.getCells().find("search", cell, opts);

	// Εάν δεν βρεθεί τέτοιο κελί, τότε σπάστε τον βρόχο
	if (cell == null)
		break;

	// Αντικαταστήστε το κελί με την τιμή αντικατάσταση
	cell.putValue("replace");

} while (true);

// Αποθηκεύστε το βιβλίο εργασίας
workbook.save(dataDir + "output.xlsm");


```
