// Tạo một sổ làm việc mới
Workbook wkb = new Workbook();

// Truy cập trang tính đầu tiên của sổ làm việc.
Worksheet worksheet = wkb.getWorksheets().get(0);

// Thêm nội dung có liên quan vào ô
worksheet.getCells().get("A1").putValue("ColumnA");
worksheet.getCells().get("B1").putValue("ColumnB")
worksheet.getCells().get("A2").putValue("ValueA")
worksheet.getCells().get("B2").putValue("ValueB")

// Lưu sổ làm việc dưới dạng tệp XLS
wkb.save("Excel.xls"); 

// Để nâng cao mã cho các chức năng khác, đây là các chức năng khác

// getCharts (). add () để thêm biểu đồ
