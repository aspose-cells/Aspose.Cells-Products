Cell cell1 = cells.Find(205, null, opts);

}else{
    Console.WriteLine("Record not found ");
}

// इनपुट स्ट्रिंग के साथ सेल खोजें

}

// इनपुट स्ट्रिंग वाले सेल का पता लगाएं
