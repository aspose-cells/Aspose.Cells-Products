string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// โหลดไฟล์ HTML ในอินสแตนซ์ของ Workbook
var book = new Aspose.Cells.Workbook("template.html");
// สร้างอินสแตนซ์ของ HtmlSaveOptions & ตั้งค่าคุณสมบัติ ExportHeadings เป็น true
var options = new Aspose.Cells.HtmlSaveOptions();
options.ExportHeadings = true;

// บันทึกไฟล์ HTML ในรูปแบบ HTML
book.Save(output, options);
// โหลด HTML ผลลัพธ์ในเบราว์เซอร์เริ่มต้น
