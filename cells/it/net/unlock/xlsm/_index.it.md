```cs

// creare un'istanza di un oggetto Workbook con un file XLSM protetto
var workbook = new Aspose.Cells.Workbook("protected.xlsm");

// accedere al foglio di lavoro predefinito nel file Excel
var worksheet = workbook.Worksheets[0];

// non proteggere il foglio di lavoro senza una password
worksheet.Unprotect();

// non proteggere la cartella di lavoro con la password
workbook.Unprotect("password");

// salvare nuovamente il risultato in formato XLSM
workbook.Save("unprotected.xlsm", Aspose.Cells.SaveFormat.Auto);


```
