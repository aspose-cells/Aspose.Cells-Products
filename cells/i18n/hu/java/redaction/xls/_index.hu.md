```cs
Workbook workbook = new Workbook(dataDir + "sourceFile.xls");

Worksheet worksheet = workbook.getWorksheets().get(0);

// Adja meg a tartományt, ahol keresni szeretne
// Itt a tartomány E3:H6
CellArea area = CellArea.createCellArea("E3", "H6");

// Adja meg a Keresés opciókat
FindOptions opts = new FindOptions();
opts.setLookInType(LookInType.VALUES);
opts.setLookAtType(LookAtType.ENTIRE_CONTENT);
opts.setRange(area);

Cell cell = null;

do {
	// Keresés a cellában értékkereséssel a tartományon belül
	cell = worksheet.getCells().find("search", cell, opts);

	// Ha nem található ilyen cella, akkor szakítsa meg a hurkot
	if (cell == null)
		break;

	// Cserélje ki a cellát a csere értékre
	cell.putValue("replace");

} while (true);

// Mentse el a munkafüzetet
workbook.save(dataDir + "output.xls");


```
