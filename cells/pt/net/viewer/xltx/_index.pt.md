{{% blocks/products/pf/agp/code-block title="C# código de exemplo para visualizar o arquivo XLTX" offSpacer="" %}}

```cs

string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// carregar o arquivo XLTX em uma instância do Workbook
var book = new Aspose.Cells.Workbook("template.xltx");
// crie uma instância de HtmlSaveOptions e defina a propriedade ExportHeadings como true
var options = new Aspose.Cells.HtmlSaveOptions();
options.ExportHeadings = true;

// salve o arquivo XLTX no formato HTML
book.Save(output, options);
// carregar HTML resultante no navegador padrão
System.Diagnostics.Process.Start(output);


```
