```cs

// Munkafüzet osztálypéldány létrehozása.
Workbook wkb = new Workbook();

// Nyissa meg a munkafüzet első munkalapját.
Worksheet sht = wkb.Worksheets[0];

// Szerezze meg a munkalap kívánt celláit.
Cell c00 = sht.Cells["A1"];
Cell c01 = sht.Cells["B1"];
Cell c10 = sht.Cells["A2"];
Cell c11 = sht.Cells["B2"];

// írja be az értéket a cellákba.
c00.PutValue("ColumnA");
c01.PutValue("ColumnB");
c10.PutValue("ValueA");
c11.PutValue("ValueB");

// Mentse el a munkafüzetet .pdf fájlként.
wkb.Save("created_one.pdf");


```
