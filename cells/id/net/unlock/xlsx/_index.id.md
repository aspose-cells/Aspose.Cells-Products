```cs

// instantiate objek Workbook dengan file XLSX yang dilindungi
var workbook = new Aspose.Cells.Workbook("protected.xlsx");

// mengakses lembar kerja default di file Excel
var worksheet = workbook.Worksheets[0];

// buka proteksi lembar kerja tanpa kata sandi
worksheet.Unprotect();

// buka proteksi buku kerja dengan kata sandi
workbook.Unprotect("password");

// simpan kembali hasilnya dalam format XLSX
workbook.Save("unprotected.xlsx", Aspose.Cells.SaveFormat.Auto);


```
