```cs

// Buat instance kelas Workbook.
Workbook wkb = new Workbook();

// Akses lembar kerja pertama dari buku kerja.
Worksheet sht = wkb.Worksheets[0];

// Dapatkan sel yang diinginkan dari lembar kerja.
Cell c00 = sht.Cells["A1"];
Cell c01 = sht.Cells["B1"];
Cell c10 = sht.Cells["A2"];
Cell c11 = sht.Cells["B2"];

// masukkan nilai ke dalam sel.
c00.PutValue("ColumnA");
c01.PutValue("ColumnB");
c10.PutValue("ValueA");
c11.PutValue("ValueB");

// Simpan Buku Kerja sebagai file .xltm.
wkb.Save("created_one.xltm");


```
