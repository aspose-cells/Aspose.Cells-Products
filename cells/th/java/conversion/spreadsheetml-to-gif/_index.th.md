```cs
// โหลดไฟล์ SPREADSHEETML ที่จะแสดงผล
Workbook workbook = new Workbook("sourceFile.spreadsheetml");
// เข้าถึงแผ่นงานเริ่มต้นจากคอลเลกชัน
Worksheet worksheet = workbook.getWorksheets().get(0);
// กำหนดพารามิเตอร์สำหรับภาพผลลัพธ์
ImageOrPrintOptions options = new ImageOrPrintOptions();
options.setOnePagePerSheet(true);
options.setImageType(ImageType.GIF);
// แปลงเวิร์กชีตเป็นรูปภาพในรูปแบบ GIF
SheetRender renderer = new SheetRender(worksheet, options);
renderer.toImage(0, "output.gif");   
   


```
