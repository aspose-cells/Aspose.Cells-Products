// 定義結果圖像的參數

options.setOnePagePerSheet(true);
options.setImageType(ImageType.PNG);
// 將工作表轉換為 PNG 格式的圖像
SheetRender renderer = new SheetRender(worksheet, options);
renderer.toImage(0, "output.png");
