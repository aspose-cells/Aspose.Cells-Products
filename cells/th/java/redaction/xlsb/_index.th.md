```cs
Workbook workbook = new Workbook(dataDir + "sourceFile.xlsb");

Worksheet worksheet = workbook.getWorksheets().get(0);

// ระบุช่วงที่คุณต้องการค้นหา
// นี่คือช่วง E3:H6
CellArea area = CellArea.createCellArea("E3", "H6");

// ระบุ ค้นหาตัวเลือก
FindOptions opts = new FindOptions();
opts.setLookInType(LookInType.VALUES);
opts.setLookAtType(LookAtType.ENTIRE_CONTENT);
opts.setRange(area);

Cell cell = null;

do {
	// ค้นหาเซลล์ด้วยการค้นหาค่าภายในช่วง
	cell = worksheet.getCells().find("search", cell, opts);

	// หากไม่พบเซลล์ดังกล่าว ให้ทำลายลูป
	if (cell == null)
		break;

	// แทนที่เซลล์ด้วยค่าแทนที่
	cell.putValue("replace");

} while (true);

// บันทึกสมุดงาน
workbook.save(dataDir + "output.xlsb");


```
