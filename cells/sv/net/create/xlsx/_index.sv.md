```cs

// Skapa Workbook-klassinstans.
Workbook wkb = new Workbook();

// Öppna det första kalkylbladet i arbetsboken.
Worksheet sht = wkb.Worksheets[0];

// Hämta önskad cell(er) i kalkylbladet.
Cell c00 = sht.Cells["A1"];
Cell c01 = sht.Cells["B1"];
Cell c10 = sht.Cells["A2"];
Cell c11 = sht.Cells["B2"];

// mata in värdet i cellen/cellerna.
c00.PutValue("ColumnA");
c01.PutValue("ColumnB");
c10.PutValue("ValueA");
c11.PutValue("ValueB");

// Spara arbetsboken som .xlsx-fil.
wkb.Save("created_one.xlsx");


```
