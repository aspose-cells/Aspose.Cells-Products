// Διαδρομή καταλόγου προέλευσης.
StringPtr srcDir = new String("SourcePath\\");

// Διαδρομή καταλόγου εξόδου.
StringPtr outDir = new String("OutputPath\\");

// Φόρτωση αρχείου CSV
intrusive_ptr<IWorkbook>  wkb = Factory::CreateIWorkbook(srcDir->StringAppend(new String("sourceFile.csv")));

// Δημιουργήστε μια παρουσία της κλάσης IReplaceOptions
intrusive_ptr<IReplaceOptions> replaceOptions = Factory::CreateIReplaceOptions();

// Ορίστε την επιλογή ευαισθησίας πεζών-κεφαλαίων
replaceOptions->SetCaseSensitive(false);

// Ορίστε την επιλογή αντιστοίχισης κειμένου
replaceOptions->SetMatchEntireCellContents(false);

// Αντικατάσταση κειμένου
wkb->Replace(new String("Text to find"), new String("Text replacement"), replaceOptions);

// Αποθήκευση ως αρχείο CSV
wkb->Save(outDir->StringAppend(new String("outputFile.csv")));
