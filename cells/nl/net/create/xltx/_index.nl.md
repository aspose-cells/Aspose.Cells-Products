```cs

// Werkmapklasse-instantie maken.
Workbook wkb = new Workbook();

// Open het eerste werkblad van de werkmap.
Worksheet sht = wkb.Worksheets[0];

// Verkrijg de gewenste cel(len) van het werkblad.
Cell c00 = sht.Cells["A1"];
Cell c01 = sht.Cells["B1"];
Cell c10 = sht.Cells["A2"];
Cell c11 = sht.Cells["B2"];

// voer de waarde in de cel(len) in.
c00.PutValue("ColumnA");
c01.PutValue("ColumnB");
c10.PutValue("ValueA");
c11.PutValue("ValueB");

// Sla de werkmap op als .xltx-bestand.
wkb.Save("created_one.xltx");


```
