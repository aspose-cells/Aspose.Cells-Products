{{% blocks/products/pf/agp/code-block title="Zdrojový kód konverze XLT na SVG Java" offSpacer="" %}}

```cs
// načtěte soubor XLT, který chcete vykreslit
Workbook workbook = new Workbook("sourceFile.xlt");
// přístup k výchozímu listu z kolekce
Worksheet worksheet = workbook.getWorksheets().get(0);
// definovat parametry pro výsledný obrázek
ImageOrPrintOptions options = new ImageOrPrintOptions();
options.setOnePagePerSheet(true);
options.setImageType(ImageType.SVG);
// převést pracovní list na obrázek ve formátu SVG
SheetRender renderer = new SheetRender(worksheet, options);
renderer.toImage(0, "output.svg");   


```
