// JSON डेटा फ़ाइल को स्ट्रिंग के रूप में पढ़ें

// एक रिक्त कार्यपुस्तिका वस्तु बनाएँ
Workbook workbook = new Workbook();
// डिफ़ॉल्ट खाली कार्यपत्रक तक पहुँचें

// स्वरूपण के लिए JsonLayoutOptions सेट करें
JsonLayoutOptions layoutOptions = new JsonLayoutOptions();
layoutOptions.setArrayAsTable(true);

// सेल A1 से शुरू होने वाली डिफ़ॉल्ट वर्कशीट में JSON डेटा आयात करें

// परिणामी छवि के लिए पैरामीटर परिभाषित करें

renderingOptions.setImageType(ImageType.JPEG);

// वर्कशीट को जेपीईजी फॉर्मेट में इमेज में बदलें
SheetRender renderer = new SheetRender(worksheet, renderingOptions);
