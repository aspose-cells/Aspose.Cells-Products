```cs

// een werkmapobject instantiëren met een beveiligd XLSX-bestand
var workbook = new Aspose.Cells.Workbook("protected.xlsx");

// toegang krijgen tot het standaardwerkblad in het Excel-bestand
var worksheet = workbook.Worksheets[0];

// werkblad deblokkeren zonder wachtwoord
worksheet.Unprotect();

// beveiliging van werkmap opheffen met wachtwoord
workbook.Unprotect("password");

// sla het resultaat terug op in XLSX-formaat
workbook.Save("unprotected.xlsx", Aspose.Cells.SaveFormat.Auto);


```
