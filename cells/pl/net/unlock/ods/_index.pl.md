// utworzyć wystąpienie obiektu Workbook z chronionym plikiem ODS
var workbook = new Aspose.Cells.Workbook("protected.ods");

// uzyskać dostęp do domyślnego arkusza roboczego w pliku Excel
var worksheet = workbook.Worksheets[0];

// odblokuj arkusz roboczy bez hasła
worksheet.Unprotect();

// odbezpiecz skoroszyt hasłem
workbook.Unprotect("password");

// zapisz wynik z powrotem w formacie ODS
workbook.Save("unprotected.ods", Aspose.Cells.SaveFormat.Auto);
