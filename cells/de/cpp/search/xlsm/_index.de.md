```cs
// Quellverzeichnispfad.
StringPtr srcDir = new String("SourcePath\\");

// Verzeichnispfad ausgeben.
StringPtr outDir = new String("OutputPath\\");

// XLSM-Datei laden
intrusive_ptr<IWorkbook>  wkb = Factory::CreateIWorkbook(srcDir->StringAppend(new String("sourceFile.xlsm")));

// Erstellen Sie eine Instanz der IReplaceOptions-Klasse
intrusive_ptr<IReplaceOptions> replaceOptions = Factory::CreateIReplaceOptions();

// Legen Sie die Option für die Groß- und Kleinschreibung fest
replaceOptions->SetCaseSensitive(false);

// Legen Sie die Textabgleichsoption fest
replaceOptions->SetMatchEntireCellContents(false);

// Texte ersetzen
wkb->Replace(new String("Text to find"), new String("Text replacement"), replaceOptions);

// Als XLSM-Datei speichern
wkb->Save(outDir->StringAppend(new String("outputFile.xlsm")));  


```
