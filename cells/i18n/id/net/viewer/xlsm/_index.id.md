string output = System.IO.Path.GetTempPath() + Guid.NewGuid().ToString() + ".html";

// muat file XLSM dalam instance Workbook
var book = new Aspose.Cells.Workbook("template.xlsm");
// buat instance HtmlSaveOptions & setel properti ExportHeadings ke true
var options = new Aspose.Cells.HtmlSaveOptions();
options.ExportHeadings = true;

// simpan file XLSM dalam format HTML
book.Save(output, options);
// memuat HTML yang dihasilkan di browser default
System.Diagnostics.Process.Start(output);
