```cs
// Vytvoření instance objektu sešitu
Workbook workbook = new Workbook(dataDir + "book1.tsv");

// Přístup k prvnímu listu v souboru TSV
Worksheet worksheet = workbook.getWorksheets().get(0);

// Hledání buňky obsahující zadaný vzorec
Cells cells = worksheet.getCells();

// Okamžitá funkce FindOptions
FindOptions findOptions = new FindOptions();

// Nalezení buňky obsahující řetězcovou hodnotu začínající na Or
findOptions.setLookAtType(LookAtType.START_WITH);

Cell cell = cells.find("SH", null, findOptions);

// Tisk názvu buňky nalezené po vyhledání 
System.out.println("Name of the cell containing String: " + cell.getName());
```
