```cs
// laad het te renderen XLTX-bestand
Workbook workbook = new Workbook("sourceFile.xltx");
// toegang tot het standaard werkblad uit de verzameling
Worksheet worksheet = workbook.getWorksheets().get(0);
// parameters definiëren voor de resulterende afbeelding
ImageOrPrintOptions options = new ImageOrPrintOptions();
options.setOnePagePerSheet(true);
options.setImageType(ImageType.PNG);
// werkblad converteren naar afbeelding in PNG-indeling
SheetRender renderer = new SheetRender(worksheet, options);
renderer.toImage(0, "output.png");   


```
