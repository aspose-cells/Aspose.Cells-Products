StringPtr outDir = new String("OutputDirectory\\");

// Charger le fichier XLSM
intrusive_ptr<IWorkbook> workbook = Factory::CreateIWorkbook(srcDir->StringAppend(new String("sampleExcelFileProtected.xlsm")));
