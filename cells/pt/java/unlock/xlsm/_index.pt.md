Workbook wkb = new Workbook("source.xlsm");

WorksheetCollection wksc = wkb.getWorksheets();
Worksheet wks = wksc.get(0);

// Unprotecting the XLSM

// Save the XLSM file.
wkb.save("Worksheet_out.xlsm", FileFormatType.EXCEL_97_TO_2003);
